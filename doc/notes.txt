FROM pg_catalog.pg_constraint c
JOIN pg_catalog.pg_class rel ON rel.oid = c.conrelid
JOIN pg_catalog.pg_namespace n ON n.oid = rel.relnamespace
WHERE n.nspname = 'public' -- Change schema if necessary AND rel.relname = 'content_tag';

6. to list total number of content grouped by sources
SELECT s.name, count(c.id) 
FROM content c
JOIN source s ON c.source = s.id 
GROUP BY s.name
ORDER BY s.name
